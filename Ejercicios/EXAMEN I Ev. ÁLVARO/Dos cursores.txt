CREATE OR REPLACE  PROCEDURE listar_tit_aut (buscar_por VARCHAR2 DEFAULT 'autor', loquebusco VARCHAR2 DEFAULT null) IS

-- Metemos en un cursor los Autores, al que le pasamos el nombre de búsqueda
CURSOR C (nombre SCOTT.AUTORES.AUTOR%TYPE, campo VARCHAR2)
IS SELECT scott.titulos.titulo,scott.autores.autor, scott.nacionalidades.nacionalidad FROM scott.autores, scott.titulos, scott.nacionalidades
WHERE scott.autores.id_nac = scott.nacionalidades.id
AND   scott.autores.id     = scott.titulos.id_aut
AND UPPER(autor) LIKE UPPER('%'||loquebusco||'%');

CURSOR C2 (nombre SCOTT.AUTORES.AUTOR%TYPE, campo VARCHAR2)
IS SELECT scott.titulos.titulo,scott.autores.autor, scott.nacionalidades.nacionalidad FROM scott.autores, scott.titulos, scott.nacionalidades
WHERE scott.autores.id_nac = scott.nacionalidades.id
AND   scott.autores.id     = scott.titulos.id_aut
AND UPPER(titulo) LIKE UPPER('%'||loquebusco||'%');

cont NUMBER(3) := 0;

texto VARCHAR2(20);

-- Inicializamos el HTML:
BEGIN

HTP.P('<html>
	<head>
		<title>Busqueda de libros</title>
	</head>

	<body>
		<h1>Buscar libros por titulo o autor</h1>

	<hr>

	<form action=listar_tit_aut method="get" >
	<b>Selecciona el campo de búsqueda</b>
		<select name="buscar_por">
			<option value=autor>Autor</option>
			<option value=titulo');
			
			-- Si la opcion a buscar es por Titulo, que lo deje seleccionado
			IF buscar_por LIKE 'titulo' THEN
				htp.p('selected="selected"');
			END IF;
			
			htp.p('>Título</option>
		</select>

	<b>Introduce el texto de busqueda</b>
		<input type="text" name="loquebusco" size="20" maxlength="30" value="'||loquebusco||'">

	<input type="submit" value="Buscar">
	</form>

	<hr>');

-- Si tenemos un texto a buscar
IF loquebusco IS NOT NULL THEN

	-- Genero la tabla
	HTP.P('<h2>Libros Localizados:</h2>
		<table width="75%" border="2" align="center">
			<tr>
				<th align="left" width=50%>Título</th>
				<th align="left" width=30%>Autor</th>
				<th align="left" width=20%>Nacionalidad</th>
			</tr>');

	IF buscar_por = 'autor' THEN
	-- Y me meto en el Cursor
	FOR C1_REC IN C(texto,buscar_por) LOOP 
		HTP.P('<TR>');
		HTP.P('<TD>'||C1_REC.titulo||'</TD><TD>'||C1_REC.autor||'</TD><TD>'||C1_REC.nacionalidad||'</TD>');
		HTP.P('</TR>');
		cont := cont + 1;
	END LOOP;
	
	ELSIF buscar_por = 'titulo' THEN
	FOR C1_REC IN C2(texto,buscar_por) LOOP 
		HTP.P('<TR>');
		HTP.P('<TD>'||C1_REC.titulo||'</TD><TD>'||C1_REC.autor||'</TD><TD>'||C1_REC.nacionalidad||'</TD>');
		HTP.P('</TR>');
		cont := cont + 1;
	END LOOP;
	
	
	END IF;
	
HTP.P('</table>

Localizados <b>'||cont||' libros</b> que se ajustan a la busqueda pedida');
END IF;
htp.p('</body></html>');
END listar_tit_aut;
/
show errors;
